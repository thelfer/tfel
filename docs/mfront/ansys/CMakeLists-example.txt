cmake_minimum_required(VERSION 3.10)

project(ansys-usermat
        DESCRIPTION "A project meant to built the usermalLib library"
        HOMEPAGE_URL https://github.com/thelfer/tfel
        LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 11)
set(CXX_STANDARD_REQUIRED ON)

include_directories("${CMAKE_CURRENT_SOURCE_DIR}")

if(WIN32)
  add_library(usermatLib SHARED usermat.cpp)
else(WIN32)
  add_library(ansuser SHARED usermat.cpp)
  if(UNIX)
    target_link_libraries(ansuser dl)
  endif(UNIX)
endif(WIN32)

add_executable(test-usermat test-usermat.cxx)
if(UNIX)
  target_link_libraries(test-usermat dl)
endif(UNIX)
